120     private static final String JDBC_DRIVER = "JDBC.driver";
121 
122     private static final String JDBC_URL = "JDBC.url";
123 
124     private final String urlTemplate;
125 
126     private final String driverClass;
127 
128     JdbcHelper(String urlTemplate, String driverClass) {
129         this.urlTemplate = urlTemplate;
130         this.driverClass = driverClass;
131     }
132 
133     public void configure(Binder binder) {
134         binder.bindConstant().annotatedWith(named(JDBC_DRIVER)).to(driverClass);
135         binder.bind(Key.get(String.class, named(JDBC_URL))).toProvider(guicify(new Formatter(urlTemplate)));
136     }
