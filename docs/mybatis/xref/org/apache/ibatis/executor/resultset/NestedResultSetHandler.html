48    public NestedResultSetHandler(Executor executor, MappedStatement mappedStatement, ParameterHandler parameterHandler, ResultHandler resultHandler, BoundSql boundSql, RowBounds rowBounds) {

55    private void ensureNoRowBounds(RowBounds rowBounds) {
56      if (rowBounds != null && (rowBounds.getLimit() < RowBounds.NO_ROW_LIMIT || rowBounds.getOffset() > RowBounds.NO_ROW_OFFSET)) {
57        throw new ExecutorException("Mapped Statements with nested result mappings cannot be safely constrained by RowBounds. "
58            + "Use safeRowBoundsEnabled=false setting to bypass this check.");
59      }
60    }

66    @Override
67    protected void handleResultSet(ResultSet rs, ResultMap resultMap, List<Object> multipleResults, ResultColumnCache resultColumnCache) throws SQLException {
68      if (resultHandler != null && configuration.isSafeResultHandlerEnabled() && !mappedStatement.isResultOrdered()) {
69        throw new ExecutorException("Mapped Statements with nested result mappings cannot be safely used with a custom ResultHandler. "
70            + "Use safeResultHandlerEnabled=false setting to bypass this check " 
71            + "or ensure your statement returns ordered data and set resultOrdered=true on it.");
72      }
73      super.handleResultSet(rs, resultMap, multipleResults, resultColumnCache);
74    }

86    @Override
87    protected void handleRowValues(ResultSet rs, ResultMap resultMap, ResultHandler resultHandler, RowBounds rowBounds, ResultColumnCache resultColumnCache) throws SQLException {
88      final DefaultResultContext resultContext = new DefaultResultContext();
89      skipRows(rs, rowBounds);
90      Object rowValue = null;
91      while (shouldProcessMoreRows(rs, resultContext, rowBounds)) {
92        final ResultMap discriminatedResultMap = resolveDiscriminatedResultMap(rs, resultMap, null);
93        final CacheKey rowKey = createRowKey(discriminatedResultMap, rs, null, resultColumnCache);
94        Object partialObject = objectCache.get(rowKey);
95        if (mappedStatement.isResultOrdered()) { // issue #577 && #542
96          if (partialObject == null && rowValue != null) {
97            objectCache.clear();
98            callResultHandler(resultHandler, resultContext, rowValue);
99          }
100         rowValue = getRowValue(rs, discriminatedResultMap, rowKey, rowKey, null, resultColumnCache, partialObject);
101       } else {
102         rowValue = getRowValue(rs, discriminatedResultMap, rowKey, rowKey, null, resultColumnCache, partialObject);
103         if (partialObject == null) {
104           callResultHandler(resultHandler, resultContext, rowValue);
105         }
106       }
107     }
108     if (rowValue != null && mappedStatement.isResultOrdered()) {
109       callResultHandler(resultHandler, resultContext, rowValue);
110     }
111   }

117   protected Object getRowValue(ResultSet rs, ResultMap resultMap, CacheKey combinedKey, CacheKey absoluteKey, String columnPrefix, ResultColumnCache resultColumnCache, Object partialObject) throws SQLException {
118     Object resultObject = partialObject;
119     if (resultObject != null) {
120       final MetaObject metaObject = configuration.newMetaObject(resultObject);
121       ancestorCache.put(absoluteKey, resultObject);
122       applyNestedResultMappings(rs, resultMap, metaObject, columnPrefix, resultColumnCache, combinedKey, false);
123       ancestorCache.remove(absoluteKey);
124     } else {
125       final ResultLoaderMap lazyLoader = instantiateResultLoaderMap();
126       resultObject = createResultObject(rs, resultMap, lazyLoader, columnPrefix, resultColumnCache);
127       if (resultObject != null && !typeHandlerRegistry.hasTypeHandler(resultMap.getType())) {
128         final MetaObject metaObject = configuration.newMetaObject(resultObject);
129         boolean foundValues = resultMap.getConstructorResultMappings().size() > 0;
130         if (shouldApplyAutomaticMappings(resultMap, AutoMappingBehavior.FULL.equals(configuration.getAutoMappingBehavior()))) {
131           final List<String> unmappedColumnNames = resultColumnCache.getUnmappedColumnNames(resultMap, columnPrefix);
132           foundValues = applyAutomaticMappings(rs, unmappedColumnNames, metaObject, columnPrefix, resultColumnCache) || foundValues;
133         }
134         final List<String> mappedColumnNames = resultColumnCache.getMappedColumnNames(resultMap, columnPrefix);
135         foundValues = applyPropertyMappings(rs, resultMap, mappedColumnNames, metaObject, lazyLoader, columnPrefix) || foundValues;
136         ancestorCache.put(absoluteKey, resultObject);
137         foundValues = applyNestedResultMappings(rs, resultMap, metaObject, columnPrefix, resultColumnCache, combinedKey, true) || foundValues;
138         ancestorCache.remove(absoluteKey);
139         foundValues = (lazyLoader != null && lazyLoader.size() > 0) || foundValues;
140         resultObject = foundValues ? resultObject : null;
141       }
142       if (combinedKey != CacheKey.NULL_CACHE_KEY) objectCache.put(combinedKey, resultObject);
143     }
144     return resultObject;
145   }

151   private boolean applyNestedResultMappings(ResultSet rs, ResultMap resultMap, MetaObject metaObject, String parentPrefix, ResultColumnCache resultColumnCache, CacheKey parentRowKey, boolean newObject) {
152     boolean foundValues = false;
153     for (ResultMapping resultMapping : resultMap.getPropertyResultMappings()) {
154       final String nestedResultMapId = resultMapping.getNestedResultMapId();
155       if (nestedResultMapId != null) {
156         try {
157           final String columnPrefix = getColumnPrefix(parentPrefix, resultMapping);
158           final ResultMap nestedResultMap = getNestedResultMap(rs, nestedResultMapId, columnPrefix);
159           final CacheKey rowKey = createRowKey(nestedResultMap, rs, columnPrefix, resultColumnCache);
160           final Object ancestorObject = ancestorCache.get(rowKey);
161           if (ancestorObject != null) { 
162             if (newObject) metaObject.setValue(resultMapping.getProperty(), ancestorObject);
163           } else {
164             final CacheKey combinedKey = combineKeys(rowKey, parentRowKey);            
165             Object rowValue = objectCache.get(combinedKey);
166             boolean knownValue = (rowValue != null);
167             final Object collectionProperty = instantiateCollectionPropertyIfAppropriate(resultMapping, metaObject);            
168             if (anyNotNullColumnHasValue(resultMapping, columnPrefix, rs)) {
169               rowValue = getRowValue(rs, nestedResultMap, combinedKey, rowKey, columnPrefix, resultColumnCache, rowValue);
170               if (rowValue != null && !knownValue) {
171                 if (collectionProperty != null) {
172                   final MetaObject targetMetaObject = configuration.newMetaObject(collectionProperty);
173                   targetMetaObject.add(rowValue);
174                 } else {
175                   metaObject.setValue(resultMapping.getProperty(), rowValue);
176                 }
177                 foundValues = true;
178               }
179             }
180           }
181         } catch (SQLException e) {
182           throw new ExecutorException("Error getting nested result map values for '" + resultMapping.getProperty() + "'.  Cause: " + e, e);
183         }
184       }
185     }
186     return foundValues;
187   }

189   private String getColumnPrefix(String parentPrefix, ResultMapping resultMapping) {
190     final StringBuilder columnPrefixBuilder = new StringBuilder();
191     if (parentPrefix != null) columnPrefixBuilder.append(parentPrefix);
192     if (resultMapping.getColumnPrefix() != null) columnPrefixBuilder.append(resultMapping.getColumnPrefix());
193     final String columnPrefix = columnPrefixBuilder.length() == 0 ? null : columnPrefixBuilder.toString().toUpperCase(Locale.ENGLISH);
194     return columnPrefix;
195   }

197   private boolean anyNotNullColumnHasValue(ResultMapping resultMapping, String columnPrefix, ResultSet rs) throws SQLException {
198     Set<String> notNullColumns = resultMapping.getNotNullColumns();
199     boolean anyNotNullColumnHasValue = true;
200     if (notNullColumns != null && !notNullColumns.isEmpty()) {
201       anyNotNullColumnHasValue = false;
202       for (String column: notNullColumns) {
203         rs.getObject(prependPrefix(column, columnPrefix));
204         if (!rs.wasNull()) {
205           anyNotNullColumnHasValue = true;
206           break;
207         }
208       }
209     }
210     return anyNotNullColumnHasValue;
211   }

213   private Object instantiateCollectionPropertyIfAppropriate(ResultMapping resultMapping, MetaObject metaObject) {
214     final String propertyName = resultMapping.getProperty();
215     Object propertyValue = metaObject.getValue(propertyName);
216     if (propertyValue == null) {
217       Class<?> type = resultMapping.getJavaType();
218       if (type == null) {
219         type = metaObject.getSetterType(propertyName);
220       }
221       try {
222         if (objectFactory.isCollection(type)) {
223           propertyValue = objectFactory.create(type);
224           metaObject.setValue(propertyName, propertyValue);
225           return propertyValue;
226         }
227       } catch (Exception e) {
228         throw new ExecutorException("Error instantiating collection property for result '" + resultMapping.getProperty() + "'.  Cause: " + e, e);
229       }
230     } else if (objectFactory.isCollection(propertyValue.getClass())) {
231       return propertyValue;
232     }
233     return null;
234   }

246   private CacheKey createRowKey(ResultMap resultMap, ResultSet rs, String columnPrefix, ResultColumnCache resultColumnCache) throws SQLException {
247     final CacheKey cacheKey = new CacheKey();
248     cacheKey.update(resultMap.getId());
249     List<ResultMapping> resultMappings = getResultMappingsForRowKey(resultMap);
250     if (resultMappings.size() == 0) {
251       if (Map.class.isAssignableFrom(resultMap.getType())) {
252         createRowKeyForMap(rs, cacheKey, resultColumnCache);
253       } else {
254         createRowKeyForUnmappedProperties(resultMap, rs, cacheKey, columnPrefix, resultColumnCache);
255       }
256     } else {
257       createRowKeyForMappedProperties(resultMap, rs, cacheKey, resultMappings, columnPrefix, resultColumnCache);
258     }
259     return cacheKey;
260   }

262   private CacheKey combineKeys(CacheKey rowKey, CacheKey parentRowKey) {
263     if (rowKey.getUpdateCount() > 1 && parentRowKey.getUpdateCount() > 1) {
264       CacheKey combinedKey;
265       try {
266         combinedKey = rowKey.clone();
267       } catch (CloneNotSupportedException e) {
268         throw new ExecutorException("Error cloning cache key.  Cause: " + e, e);
269       }
270       combinedKey.update(parentRowKey);
271       return combinedKey;
272     }
273     return CacheKey.NULL_CACHE_KEY;
274   }

276   private List<ResultMapping> getResultMappingsForRowKey(ResultMap resultMap) {
277     List<ResultMapping> resultMappings = resultMap.getIdResultMappings();
278     if (resultMappings.size() == 0) {
279       resultMappings = resultMap.getPropertyResultMappings();
280     }
281     return resultMappings;
282   }

284   private void createRowKeyForMappedProperties(
285       ResultMap resultMap, 
286       ResultSet rs, 
287       CacheKey cacheKey, 
288       List<ResultMapping> resultMappings, 
289       String columnPrefix,
290       ResultColumnCache resultColumnCache) throws SQLException {
291     for (ResultMapping resultMapping : resultMappings) {
292       if (resultMapping.getNestedResultMapId() != null) {
293         final ResultMap nestedResultMap = configuration.getResultMap(resultMapping.getNestedResultMapId());
294         createRowKeyForMappedProperties(nestedResultMap, rs, cacheKey, nestedResultMap.getConstructorResultMappings(),
295             prependPrefix(resultMapping.getColumnPrefix(), columnPrefix), resultColumnCache);
296       } else if (resultMapping.getNestedQueryId() == null) {
297         final String column = prependPrefix(resultMapping.getColumn(), columnPrefix);
298         final TypeHandler<?> th = resultMapping.getTypeHandler();
299         List<String> mappedColumnNames = resultColumnCache.getMappedColumnNames(resultMap, columnPrefix);
300         if (column != null && mappedColumnNames.contains(column.toUpperCase(Locale.ENGLISH))) { // Issue #114
301           final Object value = th.getResult(rs, column);
302           if (value != null) {
303             cacheKey.update(column);
304             cacheKey.update(value);
305           }
306         }
307       }
308     }
309   }

311   private void createRowKeyForUnmappedProperties(ResultMap resultMap, ResultSet rs, CacheKey cacheKey, String columnPrefix, ResultColumnCache resultColumnCache) throws SQLException {
312     final MetaClass metaType = MetaClass.forClass(resultMap.getType());
313     List<String> unmappedColumnNames = resultColumnCache.getUnmappedColumnNames(resultMap, columnPrefix);
314     for (String column : unmappedColumnNames) {
315       String property = column;
316       if (columnPrefix != null && columnPrefix.length() > 0) {
317         // When columnPrefix is specified,
318         // ignore columns without the prefix.
319         if (column.startsWith(columnPrefix)) {
320           property = column.substring(columnPrefix.length());
321         } else {
322           continue;
323         }
324       }
325       if (metaType.findProperty(property, configuration.isMapUnderscoreToCamelCase()) != null) {
326         String value = rs.getString(column);
327         if (value != null) {
328           cacheKey.update(column);
329           cacheKey.update(value);
330         }
331       }
332     }
333   }

335   private void createRowKeyForMap(ResultSet rs, CacheKey cacheKey, ResultColumnCache resultColumnCache) throws SQLException {
336     List<String> columnNames = resultColumnCache.getColumnNames();
337     for (String columnName : columnNames) {
338       final String value = rs.getString(columnName);
339       if (value != null) {
340         cacheKey.update(columnName);
341         cacheKey.update(value);
342       }
343     }
344   }
