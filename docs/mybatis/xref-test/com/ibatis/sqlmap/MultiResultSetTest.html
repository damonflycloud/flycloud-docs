61    public void testCallableStatementShouldReturnTwoResultSets() throws Exception {
62      sqlMap.startTransaction();
63      Connection conn = sqlMap.getCurrentConnection();
64      CallableStatement cs = conn.prepareCall("{call MRESULTSET(?,?,?,?)}");
65      cs.setInt(1, 1);
66      cs.setInt(2, 2);
67      cs.setInt(3, 3);
68      cs.setInt(4, 4);
69      cs.execute();
70      ResultSet rs = cs.getResultSet();
71      assertNotNull(rs);
72      int found = 1;
73      while (cs.getMoreResults()) {
74        assertNotNull(cs.getResultSet());
75        found++;
76      }
77      rs.close();
78      cs.close();
79      assertEquals("Didn't find second result set.", 2, found);
80    }
