76    public DaoManager buildDaoManager(Reader reader)
77        throws DaoException {
78      StandardDaoManager daoManager = new StandardDaoManager();
79  
80      try {
81  
82        Document doc = getDoc(reader);
83        Element root = (Element) doc.getLastChild();
84  
85        String rootname = root.getNodeName();
86        if (!DAO_CONFIG_ELEMENT.equals(rootname)) {
87          throw new IOException("Error while configuring DaoManager.  The root tag of the DAO configuration XML " +
88              "document must be '" + DAO_CONFIG_ELEMENT + "'.");
89        }
90  
91        NodeList children = root.getChildNodes();
92        for (int i = 0; i < children.getLength(); i++) {
93          Node child = children.item(i);
94          if (child.getNodeType() == Node.ELEMENT_NODE) {
95            if (CONTEXT_ELEMENT.equals(child.getNodeName())) {
96              DaoContext daoContext = parseContext((Element) child, daoManager);
97              daoManager.addContext(daoContext);
98            } else if (PROPERTIES_ELEMENT.equals(child.getNodeName())) {
99              Properties attributes = parseAttributes(child);
100             if (attributes.containsKey("resource")) {
101               String resource = attributes.getProperty("resource");
102               if (properties == null) {
103                 properties = Resources.getResourceAsProperties(resource);
104               } else {
105                 Properties tempProps = Resources.getResourceAsProperties(resource);
106                 tempProps.putAll(properties);
107                 properties = tempProps;
108               }
109             } else if (attributes.containsKey("url")) {
110               String url = attributes.getProperty("url");
111               if (properties == null) {
112                 properties = Resources.getUrlAsProperties(url);
113               } else {
114                 Properties tempProps = Resources.getUrlAsProperties(url);
115                 tempProps.putAll(properties);
116                 properties = tempProps;
117               }
118             }
119           }
120         }
121       }
122     } catch (Exception e) {
123       throw new DaoException("Error while configuring DaoManager.  Cause: " + e.toString(), e);
124     }
125     return daoManager;
126   }

136   private DaoContext parseContext(Element contextElement, StandardDaoManager daoManager)
137       throws DaoException {
138     DaoContext daoContext = new DaoContext();
139 
140     daoContext.setDaoManager(daoManager);
141     String id = contextElement.getAttribute("id");
142     if (id != null && id.length() > 0) {
143       daoContext.setId(id);
144     }
145 
146     NodeList children = contextElement.getChildNodes();
147     for (int i = 0; i < children.getLength(); i++) {
148       Node child = children.item(i);
149       if (child.getNodeType() == Node.ELEMENT_NODE) {
150         if (TRANS_MGR_ELEMENT.equals(child.getNodeName())) {
151           DaoTransactionManager txMgr = parseTransactionManager((Element) child);
152           daoContext.setTransactionManager(txMgr);
153         } else if (DAO_ELEMENT.equals(child.getNodeName())) {
154           DaoImpl daoImpl = parseDao((Element) child, daoManager, daoContext);
155           daoContext.addDao(daoImpl);
156         }
157       }
158     }
159 
160     return daoContext;
161   }

163   private DaoTransactionManager parseTransactionManager(Element transPoolElement)
164       throws DaoException {
165     DaoTransactionManager txMgr = null;
166 
167     Properties attributes = parseAttributes(transPoolElement);
168 
169     String implementation = attributes.getProperty("type");
170     implementation = resolveAlias(implementation);
171 
172 
173     try {
174       txMgr = (DaoTransactionManager) Resources.classForName(implementation).newInstance();
175     } catch (Exception e) {
176       throw new DaoException("Error while configuring DaoManager.  Cause: " + e.toString(), e);
177     }
178 
179     Properties props = properties;
180 
181     if (props == null) {
182       props = parsePropertyElements(transPoolElement);
183     } else {
184       props.putAll(parsePropertyElements(transPoolElement));
185     }
186 
187     txMgr.configure(props);
188 
189     if (txMgr == null) {
190       throw new DaoException("Error while configuring DaoManager.  Some unknown condition caused the " +
191           "DAO Transaction Manager to be null after configuration.");
192     }
193 
194     return txMgr;
195   }

197   private DaoImpl parseDao(Element element, StandardDaoManager daoManager, DaoContext daoContext) {
198     DaoImpl daoImpl = new DaoImpl();
199     if (element.getNodeType() == Node.ELEMENT_NODE) {
200       if (DAO_ELEMENT.equals(element.getNodeName())) {
201 
202         Properties attributes = parseAttributes(element);
203 
204         try {
205           String iface = attributes.getProperty("interface");
206           String impl = attributes.getProperty("implementation");
207           daoImpl.setDaoManager(daoManager);
208           daoImpl.setDaoContext(daoContext);
209           daoImpl.setDaoInterface(Resources.classForName(iface));
210           daoImpl.setDaoImplementation(Resources.classForName(impl));
211 
212           Class daoClass = daoImpl.getDaoImplementation();
213           Dao dao = null;
214 
215           try {
216             Constructor constructor = daoClass.getConstructor(new Class[]{DaoManager.class});
217             dao = (Dao) constructor.newInstance(new Object[]{daoManager});
218           } catch (Exception e) {
219             dao = (Dao) daoClass.newInstance();
220           }
221 
222           daoImpl.setDaoInstance(dao);
223           daoImpl.initProxy();
224         } catch (Exception e) {
225           throw new DaoException("Error configuring DAO.  Cause: " + e, e);
226         }
227       }
228     }
229     return daoImpl;
230   }

232   private Properties parsePropertyElements(Element propsParentElement) {
233 
234     Properties props = new Properties();
235 
236     NodeList children = propsParentElement.getChildNodes();
237     for (int i = 0; i < children.getLength(); i++) {
238       Node child = children.item(i);
239 
240       if (child.getNodeType() == Node.ELEMENT_NODE) {
241         if (PROPERTY_ELEMENT.equals(child.getNodeName())) {
242 
243           Properties attributes = parseAttributes(child);
244 
245           String name = attributes.getProperty("name");
246           String value = attributes.getProperty("value");
247 
248           props.setProperty(name, value);
249         }
250       }
251 
252     }
253 
254     return props;
255   }

257   private Properties parseAttributes(Node n) {
258     Properties attributes = new Properties();
259     NamedNodeMap attributeNodes = n.getAttributes();
260     for (int i = 0; i < attributeNodes.getLength(); i++) {
261       Node attribute = attributeNodes.item(i);
262       String value = parsePropertyTokens(attribute.getNodeValue());
263       attributes.put(attribute.getNodeName(), value);
264     }
265     return attributes;
266   }

268   private String parsePropertyTokens(String string) {
269     final String OPEN = "${";
270     final String CLOSE = "}";
271     String newString = string;
272     if (newString != null && properties != null) {
273       int start = newString.indexOf(OPEN);
274       int end = newString.indexOf(CLOSE);
275 
276       while (start > -1 && end > start) {
277         String prepend = newString.substring(0, start);
278         String append = newString.substring(end + CLOSE.length());
279         String propName = newString.substring(start + OPEN.length(), end);
280         String propValue = properties.getProperty(propName);
281         if (propValue == null) {
282           newString = prepend + append;
283         } else {
284           newString = prepend + propValue + append;
285         }
286         start = newString.indexOf(OPEN);
287         end = newString.indexOf(CLOSE);
288       }
289     }
290     return newString;
291   }

293   private Document getDoc(Reader reader) {
294     try {
295       // Configuration
296       DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
297       dbf.setNamespaceAware(false);
298       dbf.setValidating(true);
299       dbf.setIgnoringComments(true);
300       dbf.setIgnoringElementContentWhitespace(true);
301       dbf.setCoalescing(false);
302       dbf.setExpandEntityReferences(false);
303 
304       OutputStreamWriter errorWriter = new OutputStreamWriter(System.err);
305 
306       DocumentBuilder db = dbf.newDocumentBuilder();
307       db.setErrorHandler(new SimpleErrorHandler(new PrintWriter(errorWriter, true)));
308       db.setEntityResolver(new DaoClasspathEntityResolver());
309 
310       // Parse input file
311       Document doc = db.parse(new InputSource(reader));
312       return doc;
313     } catch (Exception e) {
314       throw new RuntimeException("XML Parser Error.  Cause: " + e);
315     }
316   }

318   private String resolveAlias(String string) {
319     String newString = null;
320     if (typeAliases.containsKey(string)) {
321       newString = (String) typeAliases.get(string);
322     }
323     if (newString != null) {
324       string = newString;
325     }
326     return string;
327   }

346     /*
347      * Returns a string describing parse exception details
348      */
349     private String getParseExceptionInfo(SAXParseException spe) {
350       String systemId = spe.getSystemId();
351       if (systemId == null) {
352         systemId = "null";
353       }
354       String info = "URI=" + systemId +
355           " Line=" + spe.getLineNumber() +
356           ": " + spe.getMessage();
357       return info;
358     }
