16  package org.apache.ibatis.type.usesjava8;
17  
18  import org.apache.ibatis.io.Resources;
19  import org.apache.ibatis.type.*;
20  import org.hamcrest.core.IsInstanceOf;
21  import org.junit.Before;
22  import org.junit.Test;
23  
24  import static org.junit.Assert.*;
25  import static org.hamcrest.core.IsInstanceOf.*;
26  
27  /**
28   * Tests for auto-detect type handlers of mybatis-typehandlers-jsr310.
29   *
30   * @author Kazuki Shimizu
31   */
32  public class Jsr310TypeHandlerRegistryTest {
33  
34    private TypeHandlerRegistry typeHandlerRegistry;
35  
36    @Before
37    public void setup() {
38      typeHandlerRegistry = new TypeHandlerRegistry();
39    }
40  
41    @Test
42    public void testFor_v1_0_0() throws ClassNotFoundException {
43      assertThat(getTypeHandler("java.time.Instant"), instanceOf(InstantTypeHandler.class));
44      assertThat(getTypeHandler("java.time.LocalDateTime"), instanceOf(LocalDateTimeTypeHandler.class));
45      assertThat(getTypeHandler("java.time.LocalDate"), instanceOf(LocalDateTypeHandler.class));
46      assertThat(getTypeHandler("java.time.LocalTime"), instanceOf(LocalTimeTypeHandler.class));
47      assertThat(getTypeHandler("java.time.OffsetDateTime"), instanceOf(OffsetDateTimeTypeHandler.class));
48      assertThat(getTypeHandler("java.time.OffsetTime"), instanceOf(OffsetTimeTypeHandler.class));
49      assertThat(getTypeHandler("java.time.ZonedDateTime"), instanceOf(ZonedDateTimeTypeHandler.class));
50    }
51  
52    @Test
53    public void testFor_v1_0_1() throws ClassNotFoundException {
54      assertThat(getTypeHandler("java.time.Month"), instanceOf(MonthTypeHandler.class));
55      assertThat(getTypeHandler("java.time.Year"), instanceOf(YearTypeHandler.class));
56    }
57  
58    @Test
59    public void testFor_v1_0_2() throws ClassNotFoundException {
60      assertThat(getTypeHandler("java.time.YearMonth"), instanceOf(YearMonthTypeHandler.class));
61      assertThat(getTypeHandler("java.time.chrono.JapaneseDate"), instanceOf(JapaneseDateTypeHandler.class));
62    }
63  
64    private TypeHandler<?> getTypeHandler(String fqcn) throws ClassNotFoundException {
65      return typeHandlerRegistry.getTypeHandler(Resources.classForName(fqcn));
66    }
67  
68  }
