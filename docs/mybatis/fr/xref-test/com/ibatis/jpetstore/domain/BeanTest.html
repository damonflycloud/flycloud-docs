16  package com.ibatis.jpetstore.domain;

18  import junit.framework.TestCase;
19  
20  import java.math.BigDecimal;
21  import java.math.BigInteger;
22  import java.util.*;
23  
24  public class BeanTest extends TestCase {
25  
26    private Class[] classes = {
27        Account.class,
28        Cart.class,
29        CartItem.class,
30        Category.class,
31        Item.class,
32        LineItem.class,
33        Order.class,
34        Product.class,
35        Sequence.class
36    };
37  
38    public void testAllReadWriteProperties() {
39      try {
40        for (int i = 0; i < classes.length; i++) {
41          Object object = classes[i].newInstance();
42          ClassIntrospector introspector = ClassIntrospector.getInstance(classes[i]);
43          List writeables = Arrays.asList(introspector.getWriteablePropertyNames());
44          List readables = Arrays.asList(introspector.getReadablePropertyNames());
45          for (int j = 0; j < writeables.size(); j++) {
46            String writeable = (String) writeables.get(j);
47            if (readables.contains(writeable)) {
48              Class type = introspector.getGetterType(writeable);
49              Object sample = getSampleFor(type);
50              BeanIntrospector probe = new BeanIntrospector();
51              probe.setObject(object, writeable, sample);
52              assertEquals(sample, probe.getObject(object, writeable));
53            }
54          }
55        }
56      } catch (Exception e) {
57        throw new RuntimeException("Error. ", e);
58      }
59    }
60  
61    public Object getSampleFor(Class type) throws Exception {
62      Map sampleMap = new HashMap();
63      sampleMap.put(String.class, "Hello");
64      sampleMap.put(Integer.class, new Integer(1));
65      sampleMap.put(int.class, new Integer(1));
66      sampleMap.put(Long.class, new Long(1));
67      sampleMap.put(long.class, new Long(1));
68      sampleMap.put(Double.class, new Double(1));
69      sampleMap.put(double.class, new Double(1));
70      sampleMap.put(Float.class, new Float(1));
71      sampleMap.put(float.class, new Float(1));
72      sampleMap.put(Short.class, new Short((short) 1));
73      sampleMap.put(short.class, new Short((short) 1));
74      sampleMap.put(Character.class, new Integer(1));
75      sampleMap.put(char.class, new Integer(1));
76      sampleMap.put(Date.class, new Date());
77      sampleMap.put(boolean.class, new Boolean(true));
78      sampleMap.put(Boolean.class, new Boolean(true));
79      sampleMap.put(BigDecimal.class, new BigDecimal("1.00"));
80      sampleMap.put(BigInteger.class, new BigInteger("1"));
81      sampleMap.put(List.class, new ArrayList());
82      sampleMap.put(List.class, new ArrayList());
83      if (!sampleMap.containsKey(type)) {
84        try {
85          sampleMap.put(type, type.newInstance());
86        } catch (Exception e) {
87          // ignore on purpose...we don't care if this fails
88        }
89      }
90      return sampleMap.get(type);
91    }
