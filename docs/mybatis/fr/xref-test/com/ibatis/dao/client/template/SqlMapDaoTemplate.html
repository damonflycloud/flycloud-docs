52    /*
53     * Gets the SQL Map Executor associated with the current
54     * DaoTransaction that this Dao is working under.  The SqlMapExecutor
55     * interface declares a number of methods for executing statements
56     * via an SqlMapClient instance.
57     *
58     * @return A SqlMapExecutor instance.
59     */
60    protected SqlMapExecutor getSqlMapExecutor() {
61      SqlMapDaoTransaction trans = (SqlMapDaoTransaction) daoManager.getTransaction(this);
62      return trans.getSqlMap();
63    }

65    /*
66     * Gets the SQL Map Transaction Manager associated with the current
67     * DaoTransaction that this Dao is working under.  The SqlMapExecutor
68     * interface declares a number of methods for executing statements
69     * via an SqlMapClient instance.
70     * <p/>
71     * NOTE: It is rare to require this in a DAO.  Only very special
72     * cases of DAO implementations will require access to the
73     * SqlMapTransactionManager.  Messing with transactions at this
74     * level might be dangerous to your data integrity (e.g. committing
75     * too early).
76     *
77     * @return A SqlMapTransactionManager instance.
78     */
79    protected SqlMapTransactionManager getSqlMapTransactionManager() {
80      SqlMapDaoTransaction trans = (SqlMapDaoTransaction) daoManager.getTransaction(this);
81      return trans.getSqlMap();
82    }

84    /*
85     * Executes a mapped SQL INSERT statement.
86     * Insert is a bit different from other update methods, as it
87     * provides facilities for returning the primary key of the
88     * newly inserted row (rather than the effected rows).  This
89     * functionality is of course optional.
90     * <p/>
91     * The parameter object is generally used to supply the input
92     * data for the INSERT values.
93     *
94     * @param id              The name of the statement to execute.
95     * @param parameterObject The parameter object (e.g. JavaBean, Map, XML etc.).
96     * @return The primary key of the newly inserted row.  This might be automatically
97     *         generated by the RDBMS, or selected from a sequence table or other source.
98     */
99    public Object insert(String id, Object parameterObject) {
100     try {
101       return getSqlMapExecutor().insert(id, parameterObject);
102     } catch (SQLException e) {
103       throw new DaoException("Failed to insert - id ["
104           + id + "], parameterObject [" + parameterObject + "]. Cause: " + e, e);
105     }
106   }
