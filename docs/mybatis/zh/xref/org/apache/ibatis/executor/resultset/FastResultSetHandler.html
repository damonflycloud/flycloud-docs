96    public void handleOutputParameters(CallableStatement cs) throws SQLException {
97      final Object parameterObject = parameterHandler.getParameterObject();
98      final MetaObject metaParam = configuration.newMetaObject(parameterObject);
99      final List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
100     for (int i = 0; i < parameterMappings.size(); i++) {
101       final ParameterMapping parameterMapping = parameterMappings.get(i);
102       if (parameterMapping.getMode() == ParameterMode.OUT || parameterMapping.getMode() == ParameterMode.INOUT) {
103         if (ResultSet.class.equals(parameterMapping.getJavaType())) {
104           handleRefCursorOutputParameter((ResultSet) cs.getObject(i + 1), parameterMapping, metaParam);
105         } else {
106           final TypeHandler<?> typeHandler = parameterMapping.getTypeHandler();
107           metaParam.setValue(parameterMapping.getProperty(), typeHandler.getResult(cs, i + 1));
108         }
109       }
110     }
111   }
112 
113   protected void handleRefCursorOutputParameter(ResultSet rs, ParameterMapping parameterMapping, MetaObject metaParam) throws SQLException {
114     try {
115       final String resultMapId = parameterMapping.getResultMapId();
116       final ResultMap resultMap = configuration.getResultMap(resultMapId);
117       final DefaultResultHandler resultHandler = new DefaultResultHandler(objectFactory);
118       final ResultColumnCache resultColumnCache = new ResultColumnCache(rs.getMetaData(), configuration);
119       handleRowValues(rs, resultMap, resultHandler, new RowBounds(), resultColumnCache);
120       metaParam.setValue(parameterMapping.getProperty(), resultHandler.getResultList());
121     } finally {
122       closeResultSet(rs); // issue #228 (close resultsets)
123     }
124   }

130   public List<Object> handleResultSets(Statement stmt) throws SQLException {
131     final List<Object> multipleResults = new ArrayList<Object>();
132     final List<ResultMap> resultMaps = mappedStatement.getResultMaps();
133     int resultMapCount = resultMaps.size();
134     int resultSetCount = 0;
135     ResultSet rs = stmt.getResultSet();
136 
137     while (rs == null) {
138       // move forward to get the first resultset in case the driver
139       // doesn't return the resultset as the first result (HSQLDB 2.1)
140       if (stmt.getMoreResults()) {
141         rs = stmt.getResultSet();
142       } else {
143         if (stmt.getUpdateCount() == -1) {
144           // no more results.  Must be no resultset
145           break;
146         }
147       }
148     }
149 
150     validateResultMapsCount(rs, resultMapCount);
151     while (rs != null && resultMapCount > resultSetCount) {
152       final ResultMap resultMap = resultMaps.get(resultSetCount);
153       ResultColumnCache resultColumnCache = new ResultColumnCache(rs.getMetaData(), configuration);
154       handleResultSet(rs, resultMap, multipleResults, resultColumnCache);
155       rs = getNextResultSet(stmt);
156       cleanUpAfterHandlingResultSet();
157       resultSetCount++;
158     }
159     return collapseSingleResultList(multipleResults);
160   }
161 
162   protected void closeResultSet(ResultSet rs) {
163     try {
164       if (rs != null) {
165         rs.close();
166       }
167     } catch (SQLException e) {
168       // ignore
169     }
170   }

172   protected void cleanUpAfterHandlingResultSet() {
173   }

175   protected void validateResultMapsCount(ResultSet rs, int resultMapCount) {
176     if (rs != null && resultMapCount < 1) {
177       throw new ExecutorException(
178           "A query was run and no Result Maps were found for the Mapped Statement '"
179               + mappedStatement.getId()
180               + "'.  It's likely that neither a Result Type nor a Result Map was specified.");
181     }
182   }

184   protected void handleResultSet(ResultSet rs, ResultMap resultMap, List<Object> multipleResults, ResultColumnCache resultColumnCache) throws SQLException {
185     try {
186       if (resultHandler == null) {
187         DefaultResultHandler defaultResultHandler = new DefaultResultHandler(objectFactory);
188         handleRowValues(rs, resultMap, defaultResultHandler, rowBounds, resultColumnCache);
189         multipleResults.add(defaultResultHandler.getResultList());
190       } else {
191         handleRowValues(rs, resultMap, resultHandler, rowBounds, resultColumnCache);
192       }
193     } finally {
194       closeResultSet(rs); // issue #228 (close resultsets)
195     }
196   }

198   protected List<Object> collapseSingleResultList(List<Object> multipleResults) {
199     if (multipleResults.size() == 1) {
200       @SuppressWarnings("unchecked")
201       List<Object> returned = (List<Object>) multipleResults.get(0);
202       return returned;
203     }
204     return multipleResults;
205   }

211   protected void handleRowValues(ResultSet rs, ResultMap resultMap, ResultHandler resultHandler, RowBounds rowBounds, ResultColumnCache resultColumnCache) throws SQLException {
212     final DefaultResultContext resultContext = new DefaultResultContext();
213     skipRows(rs, rowBounds);
214     while (shouldProcessMoreRows(rs, resultContext, rowBounds)) {
215       final ResultMap discriminatedResultMap = resolveDiscriminatedResultMap(rs, resultMap, null);
216       Object rowValue = getRowValue(rs, discriminatedResultMap, null, resultColumnCache);
217       callResultHandler(resultHandler, resultContext, rowValue);
218     }
219   }
220 
221   protected void callResultHandler(ResultHandler resultHandler, DefaultResultContext resultContext, Object rowValue) {
222     resultContext.nextResultObject(rowValue);
223     resultHandler.handleResult(resultContext);
224   }
225 
226   protected boolean shouldProcessMoreRows(ResultSet rs, ResultContext context, RowBounds rowBounds) throws SQLException {
227     return !context.isStopped() && rs.next() && context.getResultCount() < rowBounds.getLimit();
228   }
229 
230   protected void skipRows(ResultSet rs, RowBounds rowBounds) throws SQLException {
231     if (rs.getType() != ResultSet.TYPE_FORWARD_ONLY) {
232       if (rowBounds.getOffset() != RowBounds.NO_ROW_OFFSET) {
233         rs.absolute(rowBounds.getOffset());
234       }
235     } else {
236       for (int i = 0; i < rowBounds.getOffset(); i++) rs.next();
237     }
238   }

240   protected ResultSet getNextResultSet(Statement stmt) throws SQLException {
241     // Making this method tolerant of bad JDBC drivers
242     try {
243       if (stmt.getConnection().getMetaData().supportsMultipleResultSets()) {
244         // Crazy Standard JDBC way of determining if there are more results
245         if (!((!stmt.getMoreResults()) && (stmt.getUpdateCount() == -1))) {
246           return stmt.getResultSet();
247         }
248       }
249     } catch (Exception e) {
250       // Intentionally ignored.
251     }
252     return null;
253   }

259   protected Object getRowValue(ResultSet rs, ResultMap resultMap, CacheKey rowKey, ResultColumnCache resultColumnCache) throws SQLException {
260     final ResultLoaderMap lazyLoader = instantiateResultLoaderMap();
261     Object resultObject = createResultObject(rs, resultMap, lazyLoader, null, resultColumnCache);
262     if (resultObject != null && !typeHandlerRegistry.hasTypeHandler(resultMap.getType())) {
263       final MetaObject metaObject = configuration.newMetaObject(resultObject);
264       boolean foundValues = resultMap.getConstructorResultMappings().size() > 0;
265       if (shouldApplyAutomaticMappings(resultMap, !AutoMappingBehavior.NONE.equals(configuration.getAutoMappingBehavior()))) {
266         final List<String> unmappedColumnNames = resultColumnCache.getUnmappedColumnNames(resultMap, null);
267         foundValues = applyAutomaticMappings(rs, unmappedColumnNames, metaObject, null, resultColumnCache) || foundValues;
268       }
269       final List<String> mappedColumnNames = resultColumnCache.getMappedColumnNames(resultMap, null);
270       foundValues = applyPropertyMappings(rs, resultMap, mappedColumnNames, metaObject, lazyLoader, null) || foundValues;
271       foundValues = (lazyLoader != null && lazyLoader.size() > 0) || foundValues;
272       resultObject = foundValues ? resultObject : null;
273       return resultObject;
274     }
275     return resultObject;
276   }
277 
278   protected boolean shouldApplyAutomaticMappings(ResultMap resultMap, boolean def) {
279     if (resultMap.getAutoMapping() != null) return resultMap.getAutoMapping();
280     return def;
281   }

287       return null;

295   protected boolean applyPropertyMappings(ResultSet rs, ResultMap resultMap, List<String> mappedColumnNames, MetaObject metaObject, ResultLoaderMap lazyLoader, String columnPrefix) throws SQLException {
296     boolean foundValues = false;
297     final List<ResultMapping> propertyMappings = resultMap.getPropertyResultMappings();
298     for (ResultMapping propertyMapping : propertyMappings) {
299       final String column = prependPrefix(propertyMapping.getColumn(), columnPrefix);
300       if (propertyMapping.isCompositeResult() || (column != null && mappedColumnNames.contains(column.toUpperCase(Locale.ENGLISH)))) {
301         Object value = getPropertyMappingValue(rs, metaObject, propertyMapping, lazyLoader, columnPrefix);
302         final String property = propertyMapping.getProperty(); // issue #541 make property optional
303         if (value != OMIT && property != null && (value != null || isCallSettersOnNulls(metaObject.getSetterType(property)))) { // issue #377, call setter on nulls          
304             metaObject.setValue(property, value);
305             foundValues = (value != null) || foundValues;
306         }
307       }
308     }
309     return foundValues;
310   }
311 
312   protected boolean isCallSettersOnNulls(Class<?> propertyType) {
313     return (configuration.isCallSettersOnNulls() && !propertyType.isPrimitive());
314   }

316   protected Object getPropertyMappingValue(ResultSet rs, MetaObject metaResultObject, ResultMapping propertyMapping, ResultLoaderMap lazyLoader, String columnPrefix) throws SQLException {
317     if (propertyMapping.getNestedQueryId() != null) {
318       return getNestedQueryMappingValue(rs, metaResultObject, propertyMapping, lazyLoader, columnPrefix);
319     } else if (propertyMapping.getNestedResultMapId() != null) {
320       // the user added a column attribute to a nested result map, ignore it
321       return OMIT;
322     } else {
323       final TypeHandler<?> typeHandler = propertyMapping.getTypeHandler();
324       final String column = prependPrefix(propertyMapping.getColumn(), columnPrefix);
325       return typeHandler.getResult(rs, column);
326     }
327   }

329   protected boolean applyAutomaticMappings(ResultSet rs, List<String> unmappedColumnNames, MetaObject metaObject, String columnPrefix, ResultColumnCache resultColumnCache) throws SQLException {
330     boolean foundValues = false;
331     for (String columnName : unmappedColumnNames) {
332       String propertyName = columnName;
333       if (columnPrefix != null && columnPrefix.length() > 0) {
334         // When columnPrefix is specified,
335         // ignore columns without the prefix.
336         if (columnName.startsWith(columnPrefix)) {
337           propertyName = columnName.substring(columnPrefix.length());
338         } else {
339           continue;
340         }
341       }
342       final String property = metaObject.findProperty(propertyName, configuration.isMapUnderscoreToCamelCase());
343       if (property != null && metaObject.hasSetter(property)) {
344         final Class<?> propertyType = metaObject.getSetterType(property);
345         if (typeHandlerRegistry.hasTypeHandler(propertyType)) {
346           final TypeHandler<?> typeHandler = resultColumnCache.getTypeHandler(propertyType, columnName);
347           final Object value = typeHandler.getResult(rs, columnName);
348           if (value != null || isCallSettersOnNulls(propertyType)) { // issue #377, call setter on nulls
349             metaObject.setValue(property, value);
350             foundValues = (value != null) || foundValues;
351           }
352         }
353       }
354     }
355     return foundValues;
356   }

362   protected Object createResultObject(ResultSet rs, ResultMap resultMap, ResultLoaderMap lazyLoader, String columnPrefix, ResultColumnCache resultColumnCache) throws SQLException {
363     final List<Class<?>> constructorArgTypes = new ArrayList<Class<?>>();
364     final List<Object> constructorArgs = new ArrayList<Object>();
365     final Object resultObject = createResultObject(rs, resultMap, constructorArgTypes, constructorArgs, columnPrefix, resultColumnCache);
366     if (resultObject != null && configuration.isLazyLoadingEnabled() && !typeHandlerRegistry.hasTypeHandler(resultMap.getType())) {
367       final List<ResultMapping> propertyMappings = resultMap.getPropertyResultMappings();
368       for (ResultMapping propertyMapping : propertyMappings) {
369         if (propertyMapping.getNestedQueryId() != null) { // issue #109 (avoid creating proxies for leaf objects)
370           return proxyFactory.createProxy(resultObject, lazyLoader, configuration, objectFactory, constructorArgTypes, constructorArgs);
371         }
372       }
373     }
374     return resultObject;
375   }

377   protected Object createResultObject(ResultSet rs, ResultMap resultMap, List<Class<?>> constructorArgTypes, List<Object> constructorArgs, String columnPrefix, ResultColumnCache resultColumnCache)
378       throws SQLException {
379     final Class<?> resultType = resultMap.getType();
380     final List<ResultMapping> constructorMappings = resultMap.getConstructorResultMappings();
381     if (typeHandlerRegistry.hasTypeHandler(resultType)) {
382       return createPrimitiveResultObject(rs, resultMap, columnPrefix, resultColumnCache);
383     } else if (constructorMappings.size() > 0) {
384       return createParameterizedResultObject(rs, resultType, constructorMappings, constructorArgTypes, constructorArgs, columnPrefix, resultColumnCache);
385     } else {
386       return objectFactory.create(resultType);
387     }
388   }

390   protected Object createParameterizedResultObject(ResultSet rs, Class<?> resultType,
391                                                    List<ResultMapping> constructorMappings, 
392                                                    List<Class<?>> constructorArgTypes, 
393                                                    List<Object> constructorArgs, 
394                                                    String columnPrefix,
395                                                    ResultColumnCache resultColumnCache) throws SQLException {
396     boolean foundValues = false;
397     for (ResultMapping constructorMapping : constructorMappings) {
398       final Class<?> parameterType = constructorMapping.getJavaType();
399       final String column = constructorMapping.getColumn();
400       final Object value;
401       if (constructorMapping.getNestedQueryId() != null) {
402         value = getNestedQueryConstructorValue(rs, constructorMapping, columnPrefix);
403       } else if (constructorMapping.getNestedResultMapId() != null) {
404         final ResultMap resultMap = configuration.getResultMap(constructorMapping.getNestedResultMapId());
405         final ResultLoaderMap lazyLoader = instantiateResultLoaderMap();
406         value = createResultObject(rs, resultMap, lazyLoader, columnPrefix, resultColumnCache);
407       } else {
408         final TypeHandler<?> typeHandler = constructorMapping.getTypeHandler();
409         value = typeHandler.getResult(rs, prependPrefix(column, columnPrefix));
410       }
411       constructorArgTypes.add(parameterType);
412       constructorArgs.add(value);
413       foundValues = value != null || foundValues;
414     }
415     return foundValues ? objectFactory.create(resultType, constructorArgTypes, constructorArgs) : null;
416   }

418   protected Object createPrimitiveResultObject(ResultSet rs, ResultMap resultMap, String columnPrefix, ResultColumnCache resultColumnCache) throws SQLException {
419     final Class<?> resultType = resultMap.getType();
420     final String columnName;
421     if (resultMap.getResultMappings().size() > 0) {
422       final List<ResultMapping> resultMappingList = resultMap.getResultMappings();
423       final ResultMapping mapping = resultMappingList.get(0);
424       columnName = prependPrefix(mapping.getColumn(), columnPrefix);
425     } else {
426       columnName = resultColumnCache.getColumnNames().get(0);
427     }
428     final TypeHandler<?> typeHandler = resultColumnCache.getTypeHandler(resultType, columnName);
429     return typeHandler.getResult(rs, columnName);
430   }

436   protected Object getNestedQueryConstructorValue(ResultSet rs, ResultMapping constructorMapping, String columnPrefix) throws SQLException {
437     final String nestedQueryId = constructorMapping.getNestedQueryId();
438     final MappedStatement nestedQuery = configuration.getMappedStatement(nestedQueryId);
439     final Class<?> nestedQueryParameterType = nestedQuery.getParameterMap().getType();
440     final Object nestedQueryParameterObject = prepareParameterForNestedQuery(rs, constructorMapping, nestedQueryParameterType, columnPrefix);
441     Object value = null;
442     if (nestedQueryParameterObject != null) {
443       final BoundSql nestedBoundSql = nestedQuery.getBoundSql(nestedQueryParameterObject);
444       final CacheKey key = executor.createCacheKey(nestedQuery, nestedQueryParameterObject, RowBounds.DEFAULT, nestedBoundSql);
445       final Class<?> targetType = constructorMapping.getJavaType();
446       final Object nestedQueryCacheObject = getNestedQueryCacheObject(nestedQuery, key);
447       if (nestedQueryCacheObject != null && nestedQueryCacheObject instanceof List) {
448         value = resultExtractor.extractObjectFromList((List<Object>) nestedQueryCacheObject, targetType);
449       } else {
450         final ResultLoader resultLoader = new ResultLoader(configuration, executor, nestedQuery, nestedQueryParameterObject, targetType, key, nestedBoundSql);
451         value = resultLoader.loadResult();
452       }
453     }
454     return value;
455   }

457   protected Object getNestedQueryMappingValue(ResultSet rs, MetaObject metaResultObject, ResultMapping propertyMapping, ResultLoaderMap lazyLoader, String columnPrefix)
458       throws SQLException {
459     final String nestedQueryId = propertyMapping.getNestedQueryId();
460     final String property = propertyMapping.getProperty();
461     final MappedStatement nestedQuery = configuration.getMappedStatement(nestedQueryId);
462     final Class<?> nestedQueryParameterType = nestedQuery.getParameterMap().getType();
463     final Object nestedQueryParameterObject = prepareParameterForNestedQuery(rs, propertyMapping, nestedQueryParameterType, columnPrefix);
464     Object value = OMIT;
465     if (nestedQueryParameterObject != null) {
466       final BoundSql nestedBoundSql = nestedQuery.getBoundSql(nestedQueryParameterObject);
467       final CacheKey key = executor.createCacheKey(nestedQuery, nestedQueryParameterObject, RowBounds.DEFAULT, nestedBoundSql);
468       final Class<?> targetType = propertyMapping.getJavaType();
469       final Object nestedQueryCacheObject = getNestedQueryCacheObject(nestedQuery, key);
470       if (nestedQueryCacheObject != null && nestedQueryCacheObject instanceof List) {
471         value = resultExtractor.extractObjectFromList((List<Object>) nestedQueryCacheObject, targetType);
472       } else if (executor.isCached(nestedQuery, key)) {
473         executor.deferLoad(nestedQuery, metaResultObject, property, key, targetType);
474       } else {
475         final ResultLoader resultLoader = new ResultLoader(configuration, executor, nestedQuery, nestedQueryParameterObject, targetType, key, nestedBoundSql);
476         if (configuration.isLazyLoadingEnabled()) {
477           lazyLoader.addLoader(property, metaResultObject, resultLoader);
478         } else {
479           value = resultLoader.loadResult();
480         }
481       }
482     }
483     return value;
484   }

486   private Object getNestedQueryCacheObject(MappedStatement nestedQuery, CacheKey key) {
487     final Cache nestedQueryCache = nestedQuery.getCache();
488     return nestedQueryCache != null ? nestedQueryCache.getObject(key) : null;
489   }

491   protected Object prepareParameterForNestedQuery(ResultSet rs, ResultMapping resultMapping, Class<?> parameterType, String columnPrefix) throws SQLException {
492     if (resultMapping.isCompositeResult()) {
493       return prepareCompositeKeyParameter(rs, resultMapping, parameterType, columnPrefix);
494     } else {
495       return prepareSimpleKeyParameter(rs, resultMapping, parameterType, columnPrefix);
496     }
497   }

499   protected Object prepareSimpleKeyParameter(ResultSet rs, ResultMapping resultMapping, Class<?> parameterType, String columnPrefix) throws SQLException {
500     final TypeHandler<?> typeHandler;
501     if (typeHandlerRegistry.hasTypeHandler(parameterType)) {
502       typeHandler = typeHandlerRegistry.getTypeHandler(parameterType);
503     } else {
504       typeHandler = typeHandlerRegistry.getUnknownTypeHandler();
505     }
506     return typeHandler.getResult(rs, prependPrefix(resultMapping.getColumn(), columnPrefix));
507   }
508 
509   protected Object prepareCompositeKeyParameter(ResultSet rs, ResultMapping resultMapping, Class<?> parameterType, String columnPrefix) throws SQLException {
510     final Object parameterObject = instantiateParameterObject(parameterType);
511     final MetaObject metaObject = configuration.newMetaObject(parameterObject);
512     boolean foundValues = false;
513     for (ResultMapping innerResultMapping : resultMapping.getComposites()) {
514       final Class<?> propType = metaObject.getSetterType(innerResultMapping.getProperty());
515       final TypeHandler<?> typeHandler = typeHandlerRegistry.getTypeHandler(propType);
516       final Object propValue = typeHandler.getResult(rs, prependPrefix(innerResultMapping.getColumn(), columnPrefix));
517       if (propValue != null) { // issue #353 & #560 do not execute nested query if key is null
518         metaObject.setValue(innerResultMapping.getProperty(), propValue);
519         foundValues = true;
520       }
521     }
522     return foundValues ? parameterObject : null;
523   }

527       return new HashMap<Object, Object>();
528     } else {
529       return objectFactory.create(parameterType);

537   public ResultMap resolveDiscriminatedResultMap(ResultSet rs, ResultMap resultMap, String columnPrefix) throws SQLException {
538     Set<String> pastDiscriminators = new HashSet<String>();
539     Discriminator discriminator = resultMap.getDiscriminator();
540     while (discriminator != null) {
541       final Object value = getDiscriminatorValue(rs, discriminator, columnPrefix);
542       final String discriminatedMapId = discriminator.getMapIdFor(String.valueOf(value));
543       if (configuration.hasResultMap(discriminatedMapId)) {
544         resultMap = configuration.getResultMap(discriminatedMapId);
545         Discriminator lastDiscriminator = discriminator;
546         discriminator = resultMap.getDiscriminator();
547         if (discriminator == lastDiscriminator || !pastDiscriminators.add(discriminatedMapId)) {
548           break;
549         }
550       } else {
551         break;
552       }
553     }
554     return resultMap;
555   }

557   protected Object getDiscriminatorValue(ResultSet rs, Discriminator discriminator, String columnPrefix) throws SQLException {
558     final ResultMapping resultMapping = discriminator.getResultMapping();
559     final TypeHandler<?> typeHandler = resultMapping.getTypeHandler();
560     return typeHandler.getResult(rs, prependPrefix(resultMapping.getColumn(), columnPrefix));
561   }

563   protected static Set<String> prependPrefixes(Set<String> columnNames, String prefix) {
564     if (columnNames == null || columnNames.isEmpty() || prefix == null || prefix.length() == 0) {
565       return columnNames;
566     }
567     final Set<String> prefixed = new HashSet<String>();
568     for (String columnName : columnNames) {
569       prefixed.add(prependPrefix(columnName, prefix));
570     }
571     return prefixed;
572   }

574   protected static String prependPrefix(String columnName, String prefix) {
575     if (columnName == null || columnName.length() == 0) {
576       return columnName;
577     }
578     if (prefix == null || prefix.length() == 0) {
579       return columnName;
580     }
581     return (prefix + columnName);
582   }

594     protected ResultColumnCache(ResultSetMetaData metaData, Configuration configuration) throws SQLException {
595       super();
596       this.typeHandlerRegistry = configuration.getTypeHandlerRegistry();
597       final int columnCount = metaData.getColumnCount();
598       for (int i = 1; i <= columnCount; i++) {
599         columnNames.add(configuration.isUseColumnLabel() ? metaData.getColumnLabel(i) : metaData.getColumnName(i));
600         jdbcTypes.add(JdbcType.forCode(metaData.getColumnType(i)));
601         classNames.add(metaData.getColumnClassName(i));
602       }
603     }

606       return this.columnNames;
607     }

609     protected JdbcType getJdbcType(String columnName) {
610       final int index = columnNames.indexOf(columnName);
611       return jdbcTypes.get(index);
612     }

614     protected TypeHandler<?> getTypeHandler(Class<?> propertyType, String columnName) {
615       TypeHandler<?> handler = null;
616       Map<Class<?>, TypeHandler<?>> columnHandlers = typeHandlerMap.get(columnName);
617       if (columnHandlers == null) {
618         columnHandlers = new HashMap<Class<?>, TypeHandler<?>>();
619         typeHandlerMap.put(columnName, columnHandlers);
620       } else {
621         handler = columnHandlers.get(propertyType);
622       }
623       if (handler == null) {
624         handler = typeHandlerRegistry.getTypeHandler(propertyType);
625         // Replicate logic of UnknownTypeHandler#resolveTypeHandler
626         // See issue #59 comment 10
627         if (handler == null || handler instanceof UnknownTypeHandler) {
628           final int index = columnNames.indexOf(columnName);
629           final JdbcType jdbcType = jdbcTypes.get(index);
630           final Class<?> javaType = resolveClass(classNames.get(index));
631           if (javaType != null && jdbcType != null) {
632             handler = typeHandlerRegistry.getTypeHandler(javaType, jdbcType);
633           } else if (javaType != null) {
634             handler = typeHandlerRegistry.getTypeHandler(javaType);
635           } else if (jdbcType != null) {
636             handler = typeHandlerRegistry.getTypeHandler(jdbcType);
637           }
638         }
639         if (handler == null || handler instanceof UnknownTypeHandler) {
640           handler = new ObjectTypeHandler();
641         }
642         columnHandlers.put(propertyType, handler);
643       }
644       return handler;
645     }

647     private Class<?> resolveClass(String className) {
648       try {
649         final Class<?> clazz = Resources.classForName(className);
650         return clazz;
651       } catch (ClassNotFoundException e) {
652         return null;

656     private void loadMappedAndUnmappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
657       List<String> mappedColumnNames = new ArrayList<String>();
658       List<String> unmappedColumnNames = new ArrayList<String>();
659       final String upperColumnPrefix = columnPrefix == null ? null : columnPrefix.toUpperCase(Locale.ENGLISH);
660       final Set<String> mappedColumns = prependPrefixes(resultMap.getMappedColumns(), upperColumnPrefix);
661       for (String columnName : columnNames) {
662         final String upperColumnName = columnName.toUpperCase(Locale.ENGLISH);
663         if (mappedColumns.contains(upperColumnName)) {
664           mappedColumnNames.add(upperColumnName);
665         } else {
666           unmappedColumnNames.add(columnName);
667         }
668       }
669       mappedColumnNamesMap.put(getMapKey(resultMap, columnPrefix), mappedColumnNames);
670       unMappedColumnNamesMap.put(getMapKey(resultMap, columnPrefix), unmappedColumnNames);
671     }
672 
673     protected List<String> getMappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
674       List<String> mappedColumnNames = mappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
675       if (mappedColumnNames == null) {
676         loadMappedAndUnmappedColumnNames(resultMap, columnPrefix);
677         mappedColumnNames = mappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
678       }
679       return mappedColumnNames;
680     }

682     protected List<String> getUnmappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
683       List<String> unMappedColumnNames = unMappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
684       if (unMappedColumnNames == null) {
685         loadMappedAndUnmappedColumnNames(resultMap, columnPrefix);
686         unMappedColumnNames = unMappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
687       }
688       return unMappedColumnNames;
689     }
