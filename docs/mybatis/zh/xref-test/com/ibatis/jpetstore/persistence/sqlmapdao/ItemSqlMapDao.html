34    public void updateAllQuantitiesFromOrder(Order order) {
35      for (int i = 0; i < order.getLineItems().size(); i++) {
36        LineItem lineItem = (LineItem) order.getLineItems().get(i);
37        String itemId = lineItem.getItemId();
38        Integer increment = new Integer(lineItem.getQuantity());
39        Map param = new HashMap(2);
40        param.put("itemId", itemId);
41        param.put("increment", increment);
42        update("updateInventoryQuantity", param);
43      }
44    }
45  
46    public boolean isItemInStock(String itemId) {
47      Integer i = (Integer) queryForObject("getInventoryQuantity", itemId);
48      return (i != null && i.intValue() > 0);
49    }
50  
51    public PaginatedList getItemListByProduct(String productId) {
52      return queryForPaginatedList("getItemListByProduct", productId, PAGE_SIZE);
53    }
54  
55    public Item getItem(String itemId) {
56      Integer i = (Integer) queryForObject("getInventoryQuantity", itemId);
57      Item item = (Item) queryForObject("getItem", itemId);
58      item.setQuantity(i.intValue());
59      return item;
60    }
